int main(){

	char *file = "PRN17.txt";
	Satellite PRN17[1];
	ReadNoradTLE(PRN17, file);

	double JD, a, rev_sec, hr_14, epoch_14;
	char *date;
	double t[31];
	double mnmo[31], mnan[31];

	JD = jdatep(PRN17[0].refepoch);

	date = jd2dat(JD);

	printf("The Gregorian date of the TLE epoch is %s\n", date);

	printf("The Mean Anomaly is %f\n", PRN17[0].meanan);
	printf("The Mean Motion is %f\n", PRN17[0].meanmo);

	rev_sec = PRN17[0].meanmo*(1/24)*(1/60)*(1/60);

	a = pow(398600.4418/(4.0*pow(3.1415,2)*pow(rev_sec,2)),(1/3));

	printf("The semi-major axis is %f\n", a);

	hr_14 = frcofd(14,0,0);
	//epoch_14 = PRN17[0].refepoch + hr_14;
	epoch_14 = JD + hr_14;

	for(int i=0; i<32; i++){
		t[i] = epoch_14 + frcofd(0,0,(i+1)*60);
		mnmo[i] = PRN17[0].meanmo + 2*PRN17[0].ndot*(t[i]-JD) + 3*PRN17[0].nddot6*pow((t[i] - JD), 2);
		mean_anomaly_motion(&mnan[i], &mnmo[i], t[i], PRN17[0].refepoch, PRN17[0].meanan, PRN17[0].meanmo, PRN17[0].ndot, PRN17[0].nddot6);
	}

	printf("\nTime (EPsec)		N (rev/day) 		M(degrees)\n");
	for(int i=0; i<32; i++){
		printf("%f		%f		%f\n", (t[i]-JD)*60*60*24, mnmo[i], mnan[0]);
	}

	return 0;
}
